def generate_all_lines():
    lines = []
    
    # For each pair of distinct points, find all other points on the same line
    for i1 in range(5):
        for j1 in range(5):
            for i2 in range(5):
                for j2 in range(5):
                    if (i1, j1) >= (i2, j2):  # Avoid duplicates and same point
                        continue
                    
                    # Find all points collinear with (i1,j1) and (i2,j2)
                    collinear_points = [(i1, j1), (i2, j2)]
                    
                    # Direction vector
                    di = i2 - i1
                    dj = j2 - j1
                    
                    # Check all other points
                    for i3 in range(5):
                        for j3 in range(5):
                            if (i3, j3) == (i1, j1) or (i3, j3) == (i2, j2):
                                continue
                            
                            # Check if (i3,j3) is collinear with (i1,j1) and (i2,j2)
                            # Three points are collinear if cross product of vectors is 0
                            # Vector from (i1,j1) to (i2,j2): (di, dj)
                            # Vector from (i1,j1) to (i3,j3): (i3-i1, j3-j1)

                    if len(collinear_points) >= 3:
                        # Sort points for consistency
                        collinear_points.sort()
                        if collinear_points not in lines:

    return lines
